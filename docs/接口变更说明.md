## ⚠️ 需要调整的接口

### 1. AI生成游戏接口

**原接口：** `POST /api/v1/ai/generate` (普通JSON响应)

**新接口：** `POST /api/v1/ai/generate` (SSE Stream响应)

**变更原因：** 前端需要实时显示生成进度

**优先级：** 🔴 高（核心功能）

**技术实现要点：**

#### 后端实现（Node.js + Express）

```javascript
// server/routes/aiRoutes.js
const express = require('express');
const router = express.Router();

router.post('/generate', async (req, res) => {
  const { description, mode = 'create', gameId } = req.body;

  // 设置SSE响应头
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');
  res.flushHeaders(); // 立即发送响应头

  try {
    // 发送开始状态
    sendSSE(res, { type: 'status', message: '📋 正在分析描述...' });

    // 步骤1：分析描述
    await sleep(500);
    sendSSE(res, { type: 'step', step: 1, message: '识别游戏类型: 动作游戏' });
    sendSSE(res, { type: 'step', step: 2, message: '识别核心玩法: 跳跃机制' });

    // 步骤2：生成引擎
    sendSSE(res, { type: 'progress', progress: 20, message: '正在生成游戏引擎...' });
    await sleep(1000);

    // 步骤3：生成逻辑
    sendSSE(res, { type: 'progress', progress: 50, message: '正在生成游戏逻辑...' });
    await sleep(1000);

    // 步骤4：生成画面
    sendSSE(res, { type: 'progress', progress: 80, message: '正在生成游戏画面...' });
    await sleep(1000);

    // 步骤5：完成
    const gameData = await aiService.generateGame(description);
    sendSSE(res, { type: 'complete', data: gameData });

  } catch (error) {
    sendSSE(res, {
      type: 'error',
      code: 5001,
      message: error.message
    });
  } finally {
    res.end();
  }
});

// SSE发送辅助函数
function sendSSE(res, data) {
  res.write(`data: ${JSON.stringify(data)}\n\n`);
}

function sleep(ms) {
  return new Promise(resolve => setTimeout(resolve, ms));
}

module.exports = router;
```

#### 前端调用（demo.html已实现）

```javascript
async function generateGame() {
  const response = await fetch('/api/v1/ai/generate', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${token}`,
      'Content-Type': 'application/json',
      'Accept': 'text/event-stream'
    },
    body: JSON.stringify({ description })
  });

  const reader = response.body.getReader();
  const decoder = new TextDecoder();

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;

    const chunk = decoder.decode(value);
    const lines = chunk.split('\n');

    for (const line of lines) {
      if (line.startsWith('data: ')) {
        const data = JSON.parse(line.slice(6));
        handleStreamMessage(data);
      }
    }
  }
}

function handleStreamMessage(data) {
  switch (data.type) {
    case 'status':
      // 更新状态显示
      break;
    case 'step':
      // 显示步骤日志
      addLogToTerminal(data.message);
      break;
    case 'progress':
      // 更新进度条
      updateProgress(data.progress);
      break;
    case 'complete':
      // 显示完成结果
      showGameResult(data.data);
      break;
    case 'error':
      // 显示错误
      showError(data.message);
      break;
  }
}
```

### 2. AI修改游戏接口（新增）

**接口：** `POST /api/v1/ai/modify`

**用途：** 基于现有游戏进行修改生成

**优先级：** 🟡 中（编辑模式功能）

**实现要点：**

```javascript
router.post('/modify', async (req, res) => {
  const { gameId, modifyDescription } = req.body;

  // 设置SSE响应头
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');
  res.flushHeaders();

  try {
    // 加载原游戏
    sendSSE(res, { type: 'status', message: '📝 正在加载游戏配置...' });
    const originalGame = await gameService.getById(gameId);

    // 分析修改需求
    sendSSE(res, { type: 'step', step: 1, message: '分析修改需求...' });

    // 生成新版本
    sendSSE(res, { type: 'step', step: 2, message: '正在更新游戏逻辑...' });

    // 保存
    const updatedGame = await aiService.modifyGame(originalGame, modifyDescription);
    sendSSE(res, { type: 'complete', data: updatedGame });

  } catch (error) {
    sendSSE(res, { type: 'error', message: error.message });
  } finally {
    res.end();
  }
});
```
