### 1.3 设计模式

#### 分层架构原则

每一层只依赖下一层，实现解耦：

```javascript
// 页面层
import { useGameStore } from '@/store' // 只依赖Store

// Store层
import { gameApi } from '@/services'    // 只依赖Service

// Service层
import request from '@/utils/request'    // 只依赖Utils
```

#### 单一职责原则

每个模块/文件只做一件事：

- `user.js` - 只处理用户相关逻辑
- `game.js` - 只处理游戏相关逻辑
- `voice.js` - 只处理语音相关逻辑

#### 依赖注入

通过 Pinia 实现状态共享和依赖注入：

```javascript
// 任何地方都可以使用
const userStore = useUserStore()
const gameStore = useGameStore()
```

```javascript
// controllers/game.controller.js
const gameService = require('../services/game.service')
const { successResponse, errorResponse } = require('../utils/response')

exports.getMyGames = async (req, res) => {
  try {
    const userId = req.user._id
    const games = await gameService.getUserGames(userId)
    successResponse(res, games)
  } catch (error) {
    errorResponse(res, error.message, 500)
  }
}

exports.createGame = async (req, res) => {
  try {
    const userId = req.user._id
    const gameData = req.body
```

#### 服务层 (Services)

```javascript
// services/game.service.js
const Game = require('../models/game.model')
const aiService = require('./ai.service')

exports.createGame = async (userId, gameData) => {
  const game = await Game.create({
    userId,
    ...gameData,
    visibility: 'private'
  })
  return game
}

exports.publishGame = async (gameId, userId) => {
  const game = await Game.findOne({ _id: gameId, userId })
  if (!game) {
    throw new Error('游戏不存在')
  }
  game.visibility = 'public'
  await game.save()
  return game
}

exports.cloneGame = async (gameId, userId) => {
  const originalGame = await Game.findById(gameId)
  if (!originalGame || originalGame.visibility !== 'public') {
    throw new Error('无法克隆该游戏')
  }

  const clonedGame = await Game.create({
    userId,
    title: `${originalGame.title}的副本`,
    description: originalGame.description,
    code: originalGame.code,
    type: originalGame.type,
    visibility: 'private',
    isCloned: true,
    clonedFrom: gameId
  })

  return clonedGame
}
```

#### 中间件 (Middlewares)

```javascript
// middlewares/auth.middleware.js
const jwt = require('../utils/jwt')

exports.auth = async (req, res, next) => {
  try {
    const token = req.headers.authorization?.replace('Bearer ', '')

    if (!token) {
      return res.status(401).json({
        code: 401,
        message: '未登录'
      })
    }

    const decoded = jwt.verify(token)
    req.user = decoded
    next()
  } catch (error) {
    res.status(401).json({
      code: 401,
      message: '登录已过期'
    })
  }
}
```

```javascript
// services/ai.service.js
const OpenAI = require('openai')

const openai = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY

exports.generateGame = async (description) => {
  const prompt = generatePrompt(description)

  const completion = await openai.chat.completions.create({
    model: 'gpt-4',
    messages: [
      {
        role: 'system',
        content: '你是一个专业的儿童游戏开发者。'
      },
      {
        role: 'user',
        content: prompt
      }
    ],
    temperature: 0.7,
    max_tokens: 4000
  })

  const htmlCode = completion.choices[0].message.content

  // 提取HTML代码
  const codeMatch = htmlCode.match(/```html\n([\s\S]*?)\n```/) ||
                   htmlCode.match(/<html[\s\S]*<\/html>/i)

  return codeMatch ? codeMatch[1] || codeMatch[0] : htmlCode
}

      }
```

### 5.2 微信登录实现

```javascript
// services/wechat.service.js
const axios = require('axios')

exports.login = async (code) => {
  const { APPID, SECRET } = process.env

  // 调用微信API获取openid
  const response = await axios.get(
    `https://api.weixin.qq.com/sns/jscode2session?appid=${APPID}&secret=${SECRET}&js_code=${code}&grant_type=authorization_code`
  )

  const { openid, session_key, unionid } = response.data

  if (!openid) {
    throw new Error('微信登录失败')
  }

  // 查找或创建用户
  let user = await User.findOne({ openid })

  if (!user) {
    user = await User.create({ openid, unionid })
  }

  // 更新最后登录时间
  user.lastLoginAt = new Date()
  await user.save()

  return user
}
```
